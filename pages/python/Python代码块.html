# 1. 可变进度百分比

import time
import sys

# 输出进度百分比 rate_of_process
# r:当前步数，p:总步数，s:文本
def rop(r, p, s='rate of process'):
    # \r 退格到行首
    str1 = s + ': ' + str(round((r / p) * 100, 2)) + '%'
    sys.stdout.write('\r%s' % str1)
    sys.stdout.flush()
    if r == p:
        sys.stdout.write('\n')

# 4. crc校验

  1. 代码

    def crc16(x):
    x = x.split()  # 将数据按照空格分开
    a = 0xFFFF  # 初始值
    b = 0xA001  # 校验码
    for byte in x:
        a ^= int(byte, 16)
        for i in range(8):
            last = a % 2
            a >>= 1
            if last == 1:
                a ^= b
    s = hex(a).upper()

    return s[2:6]


if __name__ == "__main__":
    m = input()
    print(crc16(m))
